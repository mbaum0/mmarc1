# Accumulator <- 0
MOVI 0x00
# Accumulator <- MEM[accumulator + num_elements]
LAAD num_elem
# Branch if num_elements is 0
BIZI 0x1e

# accumulator <- value at MEM[MEM[arr_x_ptr]]
MOVI 0x00
LAAD arr_x_ptr
LAAD 0x00
STOR temp_var

# load value from y_ptr and add temp to it
MOVI 0x00
LAAD arr_y_ptr
LAAD 0x00
ADDA temp_var

# store result into temp and add to sum array
STOR temp_var
MOVI 0x00
LAAD arr_s_ptr
SAAD temp_var

# increment x_ptr by 2
MOVI 0x00
LAAD arr_x_ptr
ADDI 0x01
STOR arr_x_ptr

# increment y_ptr by 2
MOVI 0x00
LAAD arr_y_ptr
ADDI 0x01
STOR arr_y_ptr

# increment sum_ptr by 2
MOVI 0x00
LAAD arr_s_ptr
ADDI 0x01
STOR arr_s_ptr

# decrement num elements by 1
MOVI 0x00
LAAD 0x00
ADDI -0x01
STOR 0x00

# loop back
MOVI 0x00
BIZI -0x21
BIZI -0x01
